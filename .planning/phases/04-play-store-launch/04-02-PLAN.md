         String _htmlContent = '';

         @override

         Future<void> _loadDocument() async {
           final fileName = type == 'privacy'
               ? 'privacy_policy.html'
               : 'terms_of_service.html';
           final content = await rootBundle.loadString('assets/legal/$fileName');
           setState(() {
             _htmlContent = content;
             _isLoading = false;
